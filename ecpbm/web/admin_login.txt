<script type="text/javascript">
    // 重置表单（正确，保留）
    function clearForm() {
        $('#adminLoginForm').form('clear');
    }

    // 登录验证（修复后端返回布尔值true的判断）
    function checkAdminLogin() {
        $("#adminLoginForm").form("submit", {
            url: 'admininfo/login',
            method: 'post', // 明确POST方法，避免参数丢失
            onSubmit: function() {
                return $(this).form('validate'); // 表单验证正确
            },
            success: function(result) {
                var res = JSON.parse(result);
                // 关键修复：同时支持后端返回字符串'true'和布尔值true
                if (res.success === true || res.success === 'true') {
                    window.location.href = 'admin.jsp';
                    $("#adminLoginDlg").dialog("close");
                } else {
                    $.messager.show({
                        title: "登录提示",
                        msg: res.message || "登录失败，请检查用户名密码！", // 兼容无message的情况
                        timeout: 3000,
                        showType: 'slide'
                    });
                }
            },
            // 关键修复：添加错误回调，后端服务挂了也有提示
            error: function() {
                $.messager.alert('错误', '登录接口异常，请检查后端！', 'error');
            }
        });
    }
</script>
